Unterschiede von JS und Java console.log("--- 2. Unterschiede ---"); // Variablen werden ohne Datentyp deklariert mit var oder let (empfohlen) let alter = 25; // Kein expliziter Datentyp wie int oder String let name = "Anna"; console.log("Alter: " + alter + ", Name: " + name); // JS wird bei jedem Durchlauf interpretiert, Java wird einmal kompiliert // (Dies ist ein konzeptioneller Unterschied, der im Code nicht direkt gezeigt wird, // aber wichtig zu wissen ist: JavaScript-Code wird Zeile für Zeile vom Browser gelesen und ausgeführt) // Semikolon kann, muss aber nicht (aber Standard Style empfiehlt oft keine Semikolons am Ende von Zeilen, wo sie optional sind) let nachname = "Müller"; // Semikolon weggelassen (in manchen Styles erlaubt) let stadt = "Berlin"; // Semikolon gesetzt (auch erlaubt) console.log("Nachname: " + nachname + ", Stadt: " + stadt); console.log("\n"); // 3.
